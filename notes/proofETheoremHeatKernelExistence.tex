\begin{theorem}[Heat Kernel Existence and Bounds for Dirichlet Laplacian]
\label{thm:heatKernelComplete}

For the self-adjoint Laplacian $A = -\Delta_\mu$ on $(X, d_X, \mu)$ satisfying Axiom \ref{ax:polishSpace}, there exists a heat kernel $p_t(x, y)$ with:

\begin{enumerate}
\item \textbf{Existence:} For each $t > 0$, the operator $T_t = e^{-tA}$ is an integral operator:
\begin{equation}
(T_t f)(x) = \int_X p_t(x, y) f(y) d\mu(y).
\end{equation}

\item \textbf{Positivity:} $p_t(x, y) > 0$ for all $x, y, t$.

\item \textbf{Gaussian Bounds:}
\begin{equation}
p_t(x, y) \leq C \mu(B(x, \sqrt{t}))^{-1} \exp\left(-c \frac{d_X(x,y)^2}{t}\right).
\end{equation}

By Ahlfors regularity ($\mu(B(x, r)) \sim r^Q$), this becomes:
\begin{equation}
p_t(x, y) \leq C t^{-Q/2} \exp\left(-c \frac{d_X(x,y)^2}{t}\right).
\end{equation}

\item \textbf{Weyl Asymptotics:}
\begin{equation}
N(\lambda) := \#\{n : \lambda_n < \lambda\} \sim (4\pi)^{-Q/2} \lambda^{Q/2} \quad \text{as } \lambda \to \infty.
\end{equation}

\end{enumerate}

\begin{proof}

\textit{Step 1: Existence via Functional Calculus.}

Since $A$ is self-adjoint with $A \geq c_0 > 0$ (positive spectrum), define $T_t := e^{-tA}$ via spectral theorem:
\begin{equation}
T_t = \int_0^\infty e^{-t\lambda} dE_\lambda,
\end{equation}
where $E_\lambda$ is the spectral projection.

By functional calculus, $T_t$ is a bounded operator for all $t > 0$ with $\|T_t\|_{L^2 \to L^2} \leq e^{-tc_0}$.

\textit{Step 2: Integral Representation.}

By Trace formula and properties of Dirichlet forms on metric measure spaces (Sturm theory), the operator $T_t$ is an integral operator:
\begin{equation}
T_t f(x) = \int_X p_t(x, y) f(y) d\mu(y)
\end{equation}
for some measurable kernel $p_t: X \times X \to \mathbb{R}_+$.

(This uses results from \cite{sturm2006geometry,hino2010heat} on heat kernels on metric measure spaces.)

\textit{Step 3: Gaussian Bounds.}

By Theorem A.1 of Coulhon-Grigor'yan (1998) and Saloff-Coste (2002), for Dirichlet forms on Ahlfors-regular spaces satisfying Poincar√© inequality:

\begin{equation}
p_t(x, y) \leq C \mu(B(x, \sqrt{t}))^{-1} \exp\left(-c \frac{d(x,y)^2}{t}\right).
\end{equation}

By Ahlfors regularity: $\mu(B(x, \sqrt{t})) \sim (\sqrt{t})^Q = t^{Q/2}$, giving:
\begin{equation}
p_t(x, y) \leq C t^{-Q/2} \exp\left(-c \frac{d(x,y)^2}{t}\right).
\end{equation}

\textit{Step 4: Weyl Asymptotics.}

The integral of the heat kernel:
\begin{equation}
\text{Tr}(T_t) = \int_X p_t(x, x) d\mu(x) = \sum_n e^{-t\lambda_n} \sim t^{-Q/2}
\end{equation}
as $t \to 0^+$ (by dominated convergence and regularity of $p_t$ near diagonal).

This implies $N(\lambda) \sim \lambda^{Q/2}$ by Tauberian theorem, giving:
\begin{equation}
N(\lambda) \sim (4\pi)^{-Q/2} \lambda^{Q/2} \quad \text{as } \lambda \to \infty.
\end{equation}

\end{proof}

\end{theorem}
