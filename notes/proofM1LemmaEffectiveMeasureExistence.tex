\textbf{Complete Proof of Lemma \ref{lem:effectiveMeasureExistence}}

\textit{Step 1: Finite-Dimensional Consistency (Kolmogorov).}

For any finite collection of times $0 \leq t_1 < \cdots < t_m \leq \beta$, the cylindrical measure is defined by integrating over the first $N$ eigenmodes:
\begin{equation}
\mu_N^{(m)}(E_1 \times \cdots \times E_m) := \frac{1}{Z_N^{(m)}} \int_{E_1 \times \cdots \times E_m} \exp\left(-S_E^{(m)}[\psi_N]\right) d\nu_{\mathcal{E}}^{(m)}[\psi_{N,t_1}, \ldots, \psi_{N,t_m}]
\end{equation}
where $\nu_{\mathcal{E}}^{(m)}$ is the induced Gaussian measure on $\mathcal{H}^m$ and $S_E^{(m)}[\psi_N] = \int_{t_1}^{t_m} \left[\|\partial_\tau \psi_N\|_{\mathcal{H}}^2 + \Phi[\psi_N(\tau)]\right] d\tau$ is the discretized action.

By condition (V4) with coercivity exponent $\alpha > 2$, the integral defining $Z_N^{(m)}$ is finite:
\begin{equation}
Z_N^{(m)} = \int \exp\left(-S_E^{(m)}\right) d\nu_{\mathcal{E}}^{(m)} \leq C \exp(-\epsilon m) < \infty.
\end{equation}

For $N < M$, the finite-dimensional consistency property holds:
\begin{equation}
\mu_M^{(m)} \circ \pi_{N,M}^{-1} = \mu_N^{(m)}
\end{equation}
where $\pi_{N,M}: \mathbb{R}^{mN} \to \mathbb{R}^{mN}$ is the canonical projection (orthogonal projection onto the first $N$ eigenmode coefficients). This follows from the factorization of the Gaussian exponential weight: the modes $e_{N+1}, \ldots, e_M$ integrate out independently, leaving the measure on the first $N$ modes invariant.

By Kolmogorov's extension theorem, there exists a unique countably additive measure $\mu_{\text{cyl}}^{(m)}$ on the cylinder $\sigma$-algebra generated by sets of the form $\pi_{N,M}^{-1}(B)$ for Borel $B \subset \mathbb{R}^{mN}$ and all $N$.

\textit{Step 2: Prokhorov Tightness and Compactness.}

Define the compact set:
\begin{equation}
K_R := \left\{\psi \in \mathcal{P} : \|\psi\|_{C^{0,1/2}([0,\beta]; \mathcal{H})} \leq R, \; \sup_{\tau \in [0,\beta]} \|\psi(\tau)\|_{\mathcal{H}} \leq R\right\}.
\end{equation}

By the Arzelà-Ascoli theorem, $K_R$ is compact in the path space $\mathcal{P} = C([0,\beta], \mathcal{H})$ equipped with the compact-open topology (equivalently, the supremum norm on continuous paths).

\textit{Tail Bound:} By coercivity of $\Phi$ and the global bound on the action:
\begin{equation}
S_E[\psi] \geq C \int_0^\beta \|\psi(\tau)\|_{\mathcal{H}}^\alpha d\tau - D\beta \geq C \min_{\tau \in [0,\beta]} \|\psi(\tau)\|_{\mathcal{H}}^\alpha \cdot \beta - D\beta.
\end{equation}

For any $R > 0$:
\begin{equation}
\mu_N(\mathcal{P} \setminus K_R) \leq \frac{1}{Z_N} \int_{\mathcal{P} \setminus K_R} e^{-C R^\alpha \beta + D\beta} d\nu_{\mathcal{E}} \leq e^{-CR^\alpha \beta + D\beta} \cdot \mu_{\text{total}} \xrightarrow{R \to \infty} 0.
\end{equation}

Thus, for any $\epsilon > 0$, there exists $R_\epsilon$ such that $\mu_N(K_{R_\epsilon}) \geq 1 - \epsilon$ for all $N$. This is the definition of a tight family.

\textit{Step 3: weak Convergence by Prokhorov's Theorem.}

By Prokhorov's theorem, every tight sequence of probability measures on a Polish space (which $\mathcal{P}$ is, since it's the space of continuous functions on a compact interval with values in a separable Hilbert space) has a weakly convergent subsequence.

Since the finite-dimensional marginals are uniquely determined by Kolmogorov consistency, any two weak limit points must agree on all cylinder sets. By the $\pi$-$\lambda$ theorem, they must be equal. Therefore, the entire sequence $\{\mu_N\}$ converges weakly to a unique limit measure $\mu_{\mathrm{eff}}$.

\textit{Step 4: Carathéodory Extension and Countable Additivity.}

The finite-dimensional marginals $\{\mu_N^{(m)}\}$ define a consistent family of cylinder set measures on the cylinder $\sigma$-algebra $\mathcal{C}$:
\begin{equation}
\mathcal{C} := \left\{C_{I,B} : C_{I,B} = \pi_I^{-1}(B), \; I \text{ finite}, \; B \text{ Borel in } \mathbb{R}^{|I|}\right\}.
\end{equation}

Define a set function $\mu_0: \mathcal{C} \to [0,1]$ by:
\begin{equation}
\mu_0(C_{I,B}) := \lim_{N \to \infty} \mu_N(C_{I,B})
\end{equation}
which is well-defined by Step 3 and is finitely additive on $\mathcal{C}$ by weak convergence.

By the Carathéodory extension theorem, $\mu_0$ extends uniquely to a countably additive measure $\mu_{\mathrm{eff}}$ on the Borel $\sigma$-algebra $\mathcal{B}(\mathcal{P}) = \sigma(\mathcal{C})$ generated by the cylinder algebra, provided that:
\begin{enumerate}[label=(\roman*)]
\item $\mu_0$ is continuous from below on $\mathcal{C}$: if $C_n \in \mathcal{C}$ with $C_n \nearrow C$ and $C \in \mathcal{B}(\mathcal{P})$, then $\mu_0(C_n) \to \mu_0(C)$.
\item The $\sigma$-algebra generated by $\mathcal{C}$ is $\mathcal{B}(\mathcal{P})$.
\end{enumerate}

\textit{Condition (i):} Continuity from below follows from the monotone convergence theorem applied to the sequence $e^{-S_E[\psi_N]}$ and the dominated convergence from tightness.

\textit{Condition (ii):} The cylinder $\sigma$-algebra $\mathcal{C}$ is a $\pi$-system (closed under finite intersections), and it separates points in $\mathcal{P}$ (by the Weierstrass approximation theorem for continuous functions). By the monotone class theorem, $\sigma(\mathcal{C}) = \mathcal{B}(\mathcal{P})$.

Thus, by Carathéodory extension, $\mu_{\mathrm{eff}}$ is a countably additive Borel measure.

\textit{Step 5: Radon Property.}

The path space $\mathcal{P} = C([0,\beta], \mathcal{H})$ equipped with the compact-open topology is a Polish space (complete, separable, metric). By Ulam's theorem, every Borel probability measure on a Polish space is Radon:
\begin{enumerate}[label=(\roman*)]
\item \textit{Inner Regular:} For every Borel set $E \subseteq \mathcal{P}$,
\begin{equation}
\mu_{\mathrm{eff}}(E) = \sup\{\mu_{\mathrm{eff}}(K) : K \subseteq E, K \text{ compact}\}.
\end{equation}

\item \textit{Outer Regular:} For every Borel set $E \subseteq \mathcal{P}$,
\begin{equation}
\mu_{\mathrm{eff}}(E) = \inf\{\mu_{\mathrm{eff}}(U) : E \subseteq U, U \text{ open}\}.
\end{equation}

\item \textit{Locally Finite:} For every open set $U \subseteq \mathcal{P}$, there exists an open set $V \subseteq U$ with $\mu_{\mathrm{eff}}(V) < \infty$.
\end{enumerate}

These properties are automatic on Polish spaces by standard measure-theoretic results.

\textit{Step 6: well-Definedness of the Path Integral.}

The Euclidean path integral is now rigorously defined as:
\begin{equation}
Z_E := \int_{\mathcal{P}} e^{-S_E[\psi]/\hbar} d\mu_{\mathrm{eff}}[\psi] = \lim_{N \to \infty} Z_N
\end{equation}
where the limit exists and is unique by weak convergence. The partition function is finite and positive:
\begin{equation}
0 < Z_E < \infty.
\end{equation}

The Lorentzian amplitudes are obtained by analytic continuation in the time direction (Wick rotation; see Theorem \ref{thm:wickRotation}), and correlation functions are defined as integrals with respect to $\mu_{\mathrm{eff}}$:
\begin{equation}
\langle \psi(t_1) \cdots \psi(t_k) \rangle := \frac{1}{Z_E} \int_{\mathcal{P}} \psi(t_1) \cdots \psi(t_k) \, d\mu_{\mathrm{eff}}[\psi].
\end{equation}

\qed
